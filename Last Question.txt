//Memeinfo
#include<stdio.h>
#include<unistd.h>
int main()
{
     char meminfo;
        FILE *fp=fopen("/proc/meminfo","r");
        if(fp==NULL){
                fprintf(stderr,"Error opening file\n");}
        else{
                while((meminfo=fgetc(fp))!=EOF){
                printf("%c",meminfo);
}
fclose(fp);
}return 0;
}

//Cpuinfo
#include <stdio.h>

int main() {
    char cpuinfo;
    FILE *fp = fopen("/proc/xcpuinfo", "r");

    if (fp == NULL) {
        fprintf(stderr, "Error opening file\n");
    } else {
        while ((cpuinfo = fgetc(fp)) != EOF) {
            printf("%c", cpuinfo);      
        }
        fclose(fp);
    }

    return 0;
}
